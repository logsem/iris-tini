  Definition bi_log_related (Γ : list sectype) (e1 e2 : expr) (τ : sectype) :=
    ∀ Θ ρ vvs,
      envs_Persistent Θ →
      env_coherent Θ ∗
      ⟦ Γ ⟧* Θ ρ vvs ⊢ ⟦ τ ⟧ₑ Θ ρ (e1.[env_subst (vvs.*1)], e2.[env_subst (vvs.*2)]).

End bin_log_def.
